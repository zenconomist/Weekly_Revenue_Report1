Sub wr5_elozo_het_copy()
'
' wr5_elozo_het_copy WR_MASTER-ben a múlt heti adatokat értékbeillesztéssel átmásolni a múlt heti fülre
'
Call declare_variables

    Workbooks(M_FN).Activate

    Sheets("CW_akt_EUR").Select
    Columns("D:E").Select
    Selection.Copy
    Sheets("CW_elhet_EUR").Select
    Columns("D:E").Select
    Selection.PasteSpecial Paste:=xlPasteValuesAndNumberFormats, Operation:= _
        xlNone, SkipBlanks:=False, Transpose:=False
    Sheets("CWakt_U").Select
    Columns("D:E").Select
    Range("D55").Activate
    Application.CutCopyMode = False
    Selection.Copy
    Sheets("CW_elhet_U").Select
    Columns("D:E").Select
    Selection.PasteSpecial Paste:=xlPasteValuesAndNumberFormats, Operation:= _
        xlNone, SkipBlanks:=False, Transpose:=False
    Application.CutCopyMode = False
    ActiveWorkbook.Save
End Sub

Sub wr5_access_wr_export_copy()
'
'  wr5_access_wr_export_copy: WR_MASTER 3_1-es és 3_3-as füleire bemásolni az új értékeket
'

'
    Application.WindowState = xlNormal
    Windows("3_1_WR.xlsx").Activate
    Cells.Select
    Selection.Copy
    Windows(M_FN).Activate
    Sheets("3_1_WR").Select
    Cells.Select
    Selection.PasteSpecial Paste:=xlPasteValuesAndNumberFormats, Operation:= _
        xlNone, SkipBlanks:=False, Transpose:=False
    Application.WindowState = xlNormal
    Windows("3_3_WR_UNITS.xlsx").Activate
    Cells.Select
    Selection.Copy
    Windows(M_FN).Activate
    Sheets("3_3_WR_U").Select
    Cells.Select
    Selection.PasteSpecial Paste:=xlPasteValuesAndNumberFormats, Operation:= _
        xlNone, SkipBlanks:=False, Transpose:=False
    Application.CutCopyMode = False
    Windows("3_3_WR_UNITS.xlsx").Activate
    ActiveWindow.Close savechanges:=False
    Windows("3_1_WR.xlsx").Activate
    ActiveWindow.Close savechanges:=False
    ActiveWorkbook.Save
End Sub
